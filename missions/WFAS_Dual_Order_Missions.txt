		}
	}
	has_country_shield = yes
	WFAS_TNR_in_the_name_of_pope = {
		position = 2
		required_missions = {
			WFAS_TNR_the_new_order
		}
		icon = new_st_peters_basilica
		trigger = {
			religion = catholic
			if = {
				limit = {
					exists = PAP
				}
				PAP = {
					OR = {
						alliance_with = ROOT
						has_opinion = {
							who = ROOT
							value = 100
						}
					}
				}
			}
			else = {
				papal_influence = 100
			}
		}
		effect = {
			add_papal_influence = 50
			add_prestige = 20
		}
	}
	WFAS_TNR_donate_the_holy_see = {
		position = 3
		required_missions = {
			WFAS_TNR_in_the_name_of_pope
		}
		icon = dominant_economy
		trigger = {
			religion = catholic
			custom_trigger_tooltip = {
				tooltip = WFAS_TNR_dual_order_donate_holy_see.tooltip
				has_country_flag = WFAS_donated_holy_see
			}
		}
		effect = {
			palestine_area = {
				limit = {
					NOT = {
						is_permanent_claim = ROOT
					}
					NOT = {
						is_core = ROOT
					}
				}
				add_core = ROOT
			}
		}
	}
	WFAS_TNR_destroy_reformation = {
		position = 6
		required_missions = {
			WFAS_TNR_donate_the_holy_see
			WFAS_TNR_external_mission
		}
		icon = mission_rb_control_state_religion
		trigger = {
			if = {
				limit = {
					hre_size = 1
				}
				hre_religion = catholic
				hre_religion_locked = yes
			}
			else = {
				is_year = 1580
				custom_trigger_tooltip = {
					tooltip = WFAS_TNR_dual_order_destroy_reformation.tooltip
					all_neighbor_country = {
						NOT = {
							religion = protestant
						}
						NOT = {
							religion = reformed
						}
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					hre_size = 1
				}
				add_country_modifier = {
					name = "WFAS_catholic_dominance_1"
					duration = -1
				}
			}
			else = {
				add_country_modifier = {
					name = "WFAS_catholic_dominance_2"
					duration = -1
				}
			}
		}
