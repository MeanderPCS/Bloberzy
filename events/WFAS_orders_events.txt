#Treaty of Stensby (Denmark)
country_event = {
	id = WFAS_orders.6
	title = "WFAS_orders_stensby_1_title"
	desc = "WFAS_orders_stensby_1_desc"
	picture = DIPLOMACY_SIGNING_eventPicture
	fire_only_once = yes
	trigger = {
		tag = DAN
		is_year = 1450
		OR = {
			AND = {
				LIV = {
					exists = yes
					NOT = {
						is_rival = ROOT
					}
					owns_or_non_sovereign_subject_of = 35
					owns_or_non_sovereign_subject_of = 36
					owns_or_non_sovereign_subject_of = 1842
				}
				NOT = {
					is_rival = LIV
				}
				NOT = {
					war_with = LIV
				}
				IF = {
					limit = {
						LIV = {
							is_subject = yes
						}
					}
					LIV = {
						overlord = {
							NOT = {
								is_rival = ROOT
							}
						}
					}
				}
			}
			AND = {
				TNR = {
					exists = yes
					is_subject = no
					NOT = {
						is_rival = ROOT
					}
					owns_or_non_sovereign_subject_of = 35
					owns_or_non_sovereign_subject_of = 36
					owns_or_non_sovereign_subject_of = 1842
				}
				NOT = {
					war_with = LIV
				}
				NOT = {
					is_rival = TNR
				}
			}
		}
	}
	mean_time_to_happen = {
		months = 24
	}
	option = {
		name = "WFAS_orders_stensby_1_resign"
		custom_tooltip = "WFAS_orders_stensby_tooltip"
		35 = {
			owner = {
				IF = {
					limit = {
						tag = LIV
						is_subject = yes
					}
					overlord = {
						country_event = {
							id = WFAS_orders.7
						}
					}
				}
				ELSE = {
					country_event = {
						id = WFAS_orders.7
					}
				}
			}
		}
		ai_chance = {
			factor = 80
		}
	}
	option = {
		name = "WFAS_orders_stensby_1_leave"
		add_prestige = -5
		ai_chance = {
			factor = 20
		}
	}
}
